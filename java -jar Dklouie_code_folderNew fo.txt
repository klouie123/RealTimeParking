using System.Collections.Specialized;
using Microsoft.Maui.Controls.Maps;
using Microsoft.Maui.Maps;
using RealTimeParkingApp.ViewModels;

namespace RealTimeParkingApp.Views;

public partial class MapPage : ContentPage
{
    private MapViewModel vm;
    private bool isMapReady = false;

    public MapPage()
    {
        InitializeComponent();

        vm = new MapViewModel();
        BindingContext = vm;

        // Map ready handler
        map.HandlerChanged += Map_HandlerChanged;

        // Observe parking collection
        vm.Parkings.CollectionChanged += Parkings_CollectionChanged;
    }

    private void Map_HandlerChanged(object sender, EventArgs e)
    {
        if (map.Handler != null && !isMapReady)
        {
            isMapReady = true;

            // Load pins if any already in collection
            LoadMapPins();
        }
    }

    private void Parkings_CollectionChanged(object sender, NotifyCollectionChangedEventArgs e)
    {
        if (!isMapReady)
            return;

        MainThread.BeginInvokeOnMainThread(() =>
        {
            if (map == null)
                return;

            map.Pins.Clear();

            foreach (var p in vm.Parkings)
            {
                if (p == null || double.IsNaN(p.Latitude) || double.IsNaN(p.Longitude))
                    continue;

                var pin = new Pin
                {
                    Label = $"{p.Name} ({p.Distance:F2} km)",
                    Location = new Location(p.Latitude, p.Longitude)
                };

                pin.MarkerClicked += async (s, args) =>
                {
                    await Microsoft.Maui.ApplicationModel.Map.OpenAsync(
                        p.Latitude,
                        p.Longitude,
                        new MapLaunchOptions
                        {
                            NavigationMode = NavigationMode.Driving
                        });
                };

                map.Pins.Add(pin);
            }

            // Optional: move map to first parking
            if (vm.Parkings.Count > 0)
            {
                var first = vm.Parkings[0];
                map.MoveToRegion(MapSpan.FromCenterAndRadius(
                    new Location(first.Latitude, first.Longitude),
                    Distance.FromKilometers(1)));
            }
        });
    }

    private void LoadMapPins()
    {
        if (vm?.Parkings != null)
        {
            Parkings_CollectionChanged(vm.Parkings, new NotifyCollectionChangedEventArgs(NotifyCollectionChangedAction.Reset));
        }
    }
}

using System.Collections.ObjectModel;
using System.Diagnostics;
using System.Windows.Input;
using RealTimeParkingApp.Models;
using RealTimeParkingApp.Services;

namespace RealTimeParkingApp.ViewModels;

public class MapViewModel : BaseViewModel
{
    private readonly ParkingService _parkingService;
    private readonly LocationService _locationService;

    public ObservableCollection<ParkingLocation> Parkings { get; set; } = new();

    public ICommand FindParkingCommand { get; }

    public MapViewModel()
    {
        _parkingService = new ParkingService();
        _locationService = new LocationService();
        FindParkingCommand = new Command(async () => await LoadParking());
    }

    public async Task LoadParking()
    {
        try
        {
            var location = await _locationService.GetCurrentLocationAsync();
            if (location == null)
                return;

            var result = await _parkingService.GetNearestAsync(location.Latitude, location.Longitude);
            if (result == null || result.Count == 0)
                return;

            Parkings.Clear();

            foreach (var p in result)
            {
                if (p == null)
                    continue;

                if (double.IsNaN(p.Latitude) || double.IsNaN(p.Longitude))
                    continue;

                Parkings.Add(p);
            }
        }
        catch (Exception ex)
        {
            Debug.WriteLine($"LoadParking error: {ex.Message}");
        }
    }
}


        [HttpGet("nearest")]
        public async Task<IActionResult> GetNearest(double lat, double lng)
        {
            if (double.IsNaN(lat) || double.IsNaN(lng))
                return BadRequest("Invalid coordinates.");

            var parks = await _context.ParkingLocations
                .Select(p => new
                {
                    p.Id,
                    p.Name,
                    p.Latitude,
                    p.Longitude,
                    p.AllowedMinutes,
                    p.Description,
                    Distance = DistanceHelper.GetDistance(lat, lng, p.Latitude, p.Longitude)
                })
                .OrderBy(p => p.Distance)
                .Take(5)
                .ToListAsync();

            return Ok(parks);
        }
